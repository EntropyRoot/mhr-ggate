# mhr-ggate runtime dependencies
# server.py and v2ray/client_relay.py share these.
# tested with Python 3.10+ on Linux/macOS/Windows.
#
# For development / test deps, see requirements-dev.txt.

fastapi>=0.110,<1.0
uvicorn[standard]>=0.27,<1.0
httpx>=0.27,<1.0

# tomli only needed for Python < 3.11 (tomllib is stdlib in 3.11+)
tomli>=2.0; python_version < "3.11"

# certifi gives the fronted transport a sane CA bundle on systems where
# the OS store is missing or out of date. The transport works without it
# (falls back to ssl.create_default_context defaults), so it is optional.
certifi>=2024.2.2

# h2 powers the fronted transport's HTTP/2 multiplexing path
# (v2ray/h2_transport.py). Without it the relay still works — it just
# falls back to the HTTP/1.1 connection pool, which serves at most
# `pool_max` requests in parallel. Strongly recommended for the
# scMaxConcurrentPosts=100 xhttp workload that ships in the generated
# xray client config.
h2>=4.0,<5.0
